Buat tabel jika belum ada CREATE TABLE IF NOT EXISTS tables ( id_table BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, Nomor_Meja TEXT NOT NULL, Kapasitas INT DEFAULT 4, Status TEXT DEFAULT 'Tersedia', created_at TIMESTAMP DEFAULT NOW() ); CREATE TABLE IF NOT EXISTS menu ( id_menu BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, Nama_Menu TEXT NOT NULL, Harga INT NOT NULL, Status_Ketersediaan TEXT DEFAULT 'Tersedia', Kategori TEXT DEFAULT 'Makanan', Gambar TEXT, created_at TIMESTAMP DEFAULT NOW() ); CREATE TABLE IF NOT EXISTS orders ( id_order BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, id_table BIGINT REFERENCES tables(id_table) ON DELETE CASCADE, id_user BIGINT NULL, Tanggal_Order DATE NOT NULL, Waktu_Order TIME NOT NULL, Status_Order TEXT DEFAULT 'Pending', created_at TIMESTAMP DEFAULT NOW() ); CREATE TABLE IF NOT EXISTS orderdetails ( id_order_detail BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, id_order BIGINT REFERENCES orders(id_order) ON DELETE CASCADE, id_menu BIGINT REFERENCES menu(id_menu) ON DELETE CASCADE, Jumlah INT DEFAULT 1, Harga_Satuan INT NOT NULL, Subtotal INT NOT NULL, Note TEXT, created_at TIMESTAMP DEFAULT NOW() ); CREATE TABLE IF NOT EXISTS users ( id_user BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, Nama TEXT NOT NULL, Password TEXT NOT NULL, Role TEXT DEFAULT 'Customer', created_at TIMESTAMP DEFAULT NOW() ); -- 2.
